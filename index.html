Les entiers naturels


data Nat = Z | S Nat

plus : Nat -> Nat -> Nat
plus Z y = y
plus (S k) y = S (plus k y)

Les vecteurs


data Vect : Nat -> Type -> Type where
  Nil  : Vect Z a
  (::) : (x : a) -> (xs : Vect n a) -> Vect (S n) a

head : Vect (S n) a -> a
head (x::xs) = x

(++) : Vect m a -> Vect n a -> Vect (m + n) a
(++) []      ys = ys
(++) (x::xs) ys = x :: xs ++ ys

index 2 [1,2,3,4] -- compile et renvoie 3
